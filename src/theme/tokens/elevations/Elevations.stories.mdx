# Elevations

Elevations são sombras e desfoques que geram um efeito em camadas que dá aos usuários uma indicação visual de que um objeto ou experiência está "elevado" em prioridade ou hierarquia.

export const elevationLevels = [
  { level: 'none', text: 'Base' },
  { level: 'inner', text: 'Input' },
  { level: 'xxs', text: 'Cards' },
  { level: 'xs', text: 'Button' },
  { level: 's', text: 'Navigation Menu' },
  { level: 'm', text: 'Card Raised' },
  { level: 'l', text: 'Picker' },
  { level: 'xl', text: 'Modals' },
];

<Story name="Elevations">
  <GridBackground padding={{ xs: 's' }} justifyContent={{ xs: 'space-between' }}>
    {elevationLevels.map((item, index) => {
      const { level, text } = item;
      return (
        <Column xs={12} s={6} m={3} padding={{ xs: 's' }} key={`elevations-${index - 1}`}>
          <ViewBorder
            inline={{ xs: false }}
            padding={{ xs: 'l', m: 'xl' }}
            style={{ boxShadow: elevationsToken[level] }}
          >
            <Text as="b">Elevation {level}</Text>
            <Text>{text}</Text>
          </ViewBorder>
        </Column>
      );
    })}
  </GridBackground>
</Story>
