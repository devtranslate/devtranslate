### Body

export const renderTHead = () => {
  return (
    <tr>
      <th>Nome</th>
      <th>Font Size</th>
      <th>Line Height</th>
      <th>Exemplo</th>
    </tr>
  );
};

export const renderTBodyText = () => {
  const bodyFontSize = typographyToken.fontSizes.body;
  const bodyLineHeight = typographyToken.lineHeights.body;
  const typographyFontSizes = Object.keys(bodyFontSize);
  return typographyFontSizes.map((item, index) => {
    return (
      <tr key={`body-${index - 1}`}>
        <td>{item}</td>
        <td>{bodyFontSize[item]}</td>
        <td>{bodyLineHeight[item]}</td>
        <td>
          <Text size={{ xs: item }}>Text</Text>
        </td>
      </tr>
    );
  });
};

<Story name="Body">
  <Table>
    <thead>{renderTHead()}</thead>
    <tbody>{renderTBodyText()}</tbody>
  </Table>
</Story>

### Heading Default

export const renderTBodyHeading = (variant) => {
  const headingFontSize = typographyToken.fontSizes.heading[variant];
  const headingLineHeight = typographyToken.lineHeights.heading[variant];
  const typographyFontSizes = Object.keys(headingFontSize);
  return typographyFontSizes.map((item, index) => {
    return (
      <tr key={`heading-${index - 1}`}>
        <td>{item}</td>
        <td>{headingFontSize[item]}</td>
        <td>{headingLineHeight[item]}</td>
        <td>
          <Heading variant={variant} size={{ xs: item }} textTransform="capitalize">
            {variant === 'default' ? 'Heading' : variant}
          </Heading>
        </td>
      </tr>
    );
  });
};

<Story name="Heading (Default)">
  <Table>
    <thead>{renderTHead()}</thead>
    <tbody>{renderTBodyHeading('default')}</tbody>
  </Table>
</Story>
